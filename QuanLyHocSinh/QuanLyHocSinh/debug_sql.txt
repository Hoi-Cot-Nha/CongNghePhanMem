-- 1. Check Lop data
SELECT DISTINCT MaLop, NienKhoa FROM Lop ORDER BY MaLop;

-- 2. Check HocSinh data for L10A1 and L11A1
SELECT COUNT(*) AS [Số HS L10A1] FROM HocSinh WHERE MaLop = 'L10A1';
SELECT COUNT(*) AS [Số HS L11A1] FROM HocSinh WHERE MaLop = 'L11A1';

-- 3. Check HocPhi data - check NamHoc format
SELECT DISTINCT NamHoc FROM HocPhi ORDER BY NamHoc;

-- 4. Check HocPhi for L10A1
SELECT hp.*, hs.MaLop 
FROM HocPhi hp
JOIN HocSinh hs ON hp.MaHS = hs.MaHS
WHERE hs.MaLop = 'L10A1' 
  AND hp.HocKy = 1 
  AND hp.NamHoc = '2023-2026';

-- 5. Check HocPhi for L11A1  
SELECT hp.*, hs.MaLop
FROM HocPhi hp
JOIN HocSinh hs ON hp.MaHS = hs.MaHS
WHERE hs.MaLop = 'L11A1'
  AND hp.HocKy = 1 
  AND hp.NamHoc = '2023-2026';

-- 6. Check all HocPhi with HocSinh join
SELECT TOP 20 hp.MaHP, hp.MaHS, hs.MaLop, hp.HocKy, hp.NamHoc, hs.HoTen
FROM HocPhi hp
JOIN HocSinh hs ON hp.MaHS = hs.MaHS
ORDER BY hs.MaLop, hp.HocKy, hp.NamHoc;
